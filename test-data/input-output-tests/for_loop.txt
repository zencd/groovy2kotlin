    }

            println(xxx)
        }
    }

    fun while_loop() {
        while (false) {
        }
    }

    fun for_ever() {
        for (; ; ) {
        }
    }

    fun for3() {
        for (val i: Int = 0; i < 3; i++) {
        }
    }

    fun loop_continue() {
        while (false) {
            continue
        }
    }

    fun loop_continue_labeled() {
        @label
        while (false) {
            continue@label
        }
    }

    fun loop_break() {
        while (false) {
            break
        }
    }

    fun loop_break_labeled() {
        @label
        while (false) {
            break@label
        }
    }
